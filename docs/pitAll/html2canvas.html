<script>
    function startDownIamge() {
        let that = this
        let list = []
        if (type === 'all') {
        //测试最大30个canvas,子根据实际情况测出值，因为超过的话会报各种找不到的问题
            for (let i = 0; i < 10; i++) {
                list = this.cardData.concat(list)
            }
        } else {
            list = [item]
        }
        
        that.$set(that, 'showCardData', list)
        that.$nextTick(() => {
            var imgData = [];
            let imageDomLen = list.length
        
            for (let i = 0; i < imageDomLen; i++) {
                let dom = this.$refs[`downCanvasBox${i}`][0]
                //html2canvas每次都会重写请求所有的img资源这个比较耗时
                html2canvas(dom, {
                    //***原生宽高也可以解决只能截取到可视区的问题***
                    width:dom.offsetWidth,//dom 原始宽度
                    height:dom:offsetHeight//dom 原始高度
                    useCORS: true, // 允许图片跨域
                }).then((canvas) => {
                    imgData.push(canvas.toDataURL())
                    if (imgData.length === imageDomLen) {
                        // 下载图片包
                        that.download(imgData)
                    }
                })
        
            }
        
        
        })
    }


</script>

# jszip下载图片包

<script src="https://cdn.bootcss.com/jszip/3.2.2/jszip.min.js"></script>
<script src="https://cdn.bootcss.com/FileSaver.js/1.3.8/FileSaver.min.js"></script>
<script >
    function download(imgData) {
        var zip = new JSZip();
        for (let i = 0; i < imgData.length; i++) {
            let item = imgData[i]
            var img_arr = item.split(',');
            //根据base64数据在压缩包中生成jpg数据
            zip.file(i.toString() + '.jpg', img_arr[1], {base64: true});
        }
    
        zip.generateAsync({type: "blob"}).then(function (content) {
            saveAs(content, 'aa.zip');
        });
    }
</script>
