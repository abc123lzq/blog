 sum=x+y;
 return sum;

# 函数

函数实际上是对象，因此函数名 是一个指向函数对象的指针

function  sum(x,y) {
    return x + y;
}
console.log(sum(10, 10));//调用函数

var a = sum;  //不带圆括号的函数名是访问函数的指针，而非调用函数
console.log(a(10, 10)); //a和sum同时指向了同一个函数

sum = null;  // 尝试让sum指针与函数断绝关系
console.log(a(10,10)); //因为a指针自己有保留函数对象的地址，所以仍然可以调用函数

参数

函数体内可通过arguments来访问 参数组对象

function a() {
    if(arguments.length==1) {
       console.log(arguments[0] + 10);

       console.log(arguments[0]+arguments[1])
    }
}

a(10);  //20
a(30, 10);  //40

# DOM

DOM可以将HTML和XML文档描绘成一个由多层节点构成的结构。

节点之间的关系构成了 层次，而所有页面标记则表现为一个以特定节点为根节点的树形结构。

文档只有一个子节点，即html元素，文档节点是每个文档的根节点。

# 元素特性

每个元素都有一个或多个特性，操作特性的DOM方法有三种： getAttribute()、setAttribute()、removeAttribute()

getAttribute()

var div = document.getElementById("myDiv");
console.log(div.getAttribute("id"));

setAttribute()

接收两个参数：（要设置的特性名，值）,如果特性已存在就替换值，如果不存在直接创建并设置值

var div = document.getElementById("myDiv");
div.setAttribute("class", "lzq");
console.log(div.id);

removeAttribute()

var div = document.getElementById("myDiv");
div.removeAttribute("class", "lzq");
console.log(div.id);

# 创建元素

var div = document.createElement("div");
div.id = "myDiv";
div.className = "myClass";

document.body.appendChild(div);

# DOM扩展

HTML5新增了大量javascriptAPI

querySelector()方法

var body = document.querySelector("body");

var selected = document.querySelector(".selected");

querySelectorAll方法

var ems = document.getElementById("myDiv").querySelectorAll("em");

var selecteds=document.querySelectorAll(".selected");

getElementsByClassName()方法

var a = document.getElementsByClassName("myName yourName");

readyState 属性

loading: 正在加载文档 ,complete: 已经加载完文档

if(document.readState=="complete"){
    console.log("文档加载完毕")
}
