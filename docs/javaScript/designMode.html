# 设计模式

设计模式主要的思路，而不是样板模式，所以业务的实现流程可自由组装编写

# 单例模式

利用闭包存储私有变量来达到只执行一次的控制效果，vue的once就是这原理

 function once (fn) {
  let called = false //私有变量
  return function () {
    if (!called) {
      called = true
      fn.apply(this, arguments)
    }else {
        console.log("不在执行")
    }
  }
}
//调用
var fn=once(()=>{console.log(123)})
once() //123
once()// 不在执行
once()// 不在执行

# 观察者模式

通俗点就是你订阅主播做开播去观看事件，主播开播的时候通知你开去观看了，vue的事件总线就是这原理











 








 









 
 
 













 
 
 
 
 



















/**
 * 发布订阅模式(观察者模式)
 * handles: 事件处理函数集合
 * on: 订阅事件
 * emit: 发布事件
 * off: 删除事件
**/

class PubSub {
  constructor() {
    this.handles = {}; //事件总线
  }

  // 订阅事件
  on(eventType, handle) {
    if (!this.handles.hasOwnProperty(eventType)) {
      this.handles[eventType] = [];
    }
    if (typeof handle == 'function') {
      this.handles[eventType].push(handle); //扔进去
    } else {
      throw new Error('缺少回调函数');
    }
    return this;
  }

  // 发布事件
  emit(eventType, ...args) {
    if (this.handles.hasOwnProperty(eventType)) {
      this.handles[eventType].forEach((item, key, arr) => {
        item.apply(null, args); //因为需要两个参数，如果不需要改变this，就用null填充
      }) //遍历执行
    } else {
      throw new Error(`"${eventType}"事件未注册`);
    }
    return this;
  }

  // 删除事件
  off(eventType, handle) {
    if (!this.handles.hasOwnProperty(eventType)) {
      throw new Error(`"${eventType}"事件未注册`);
    } else if (typeof handle != 'function') {
      throw new Error('缺少回调函数');
    } else {
      this.handles[eventType].forEach((item, key, arr) => {
        if (item == handle) {
          arr.splice(key, 1);
        }
      })//遍历删除
    }
    return this; // 实现链式操作
  }
}

//调用
let callback = function () {
  console.log('6666秀秀秀');
}

let pubsub = new PubSub();
pubsub.on('completed', (...args) => {
  console.log(args.join(' '));
}).on('completed',callback)

pubsub.emit('completed', 'pubsub', '注册了两个', 'completed事件');
pubsub.off('completed', callback); //只会清除对应的回调函数
pubsub.emit('completed', 'pubsub', 'callback对应的事件被清除了');

# 职责

函数一等公民，而纯函数职责分明，从而能达到最大复用，也用防止普通函数里的逻辑混乱不清晰。

const order500 = function(orderType, pay, stock) {
  if ( orderType === 1 && pay === true ) {
    console.log('500 元定金预购, 得到 100 元优惠券')
  } else {
    order200(orderType, pay, stock)
  }
}

const order200 = function(orderType, pay, stock) {
  if ( orderType === 2 && pay === true ) {
    console.log('200 元定金预购, 得到 50 元优惠券')
  } else {
    orderCommon(orderType, pay, stock)
  }
}

const orderCommon = function(orderType, pay, stock) {
  if (orderType === 3 && stock > 0) {
    console.log('普通购买, 无优惠券')
  } else {
    console.log('库存不够, 无法购买')
  }
}

现在函数职责比较清晰，链路代码和业务代码依然耦合在一起

# 链路

引用了AOP(面向切面的编程)思想

const order500 = function(orderType, pay, stock) {
  if ( orderType === 1 && pay === true ) {
    console.log('500 元定金预购, 得到 100 元优惠券')
  } else {
    return 'nextSuccess'
  }
}

const order200 = function(orderType, pay, stock) {
  if ( orderType === 2 && pay === true ) {
    console.log('200 元定金预购, 得到 50 元优惠券')
  } else {
    return 'nextSuccess'
  }
}

const orderCommon = function(orderType, pay, stock) {
  if (orderType === 3 && stock > 0) {
    console.log('普通购买, 无优惠券')
  } else {
    console.log('库存不够, 无法购买')
  }
}

// 链路代码
Function.prototype.after = function(fn) {
  const self = this
  return function() {
    const result = self.apply(self, arguments)
    if (result === 'nextSuccess') {
      return fn.apply(self, arguments) // return 函数就能链路调用了
    }
  }
}

const order = order500.after(order200).after(orderCommon)

order( 3, true, 500 ) // 普通购买, 无优惠券

Function.prototype.before = function(fn) {
 const self = this
  return function() {
    fn.apply(self,arguments);//先执行before的回调
    self.apply(this,arguments); //执行原有函数
  } 
}

Function.prototype.after = function(fn) {
 const self = this
  return function() {
    self.apply(this,arguments); //执行当前函数
    fn.apply(this,arguments);   //再执行after的回调

  }
}
