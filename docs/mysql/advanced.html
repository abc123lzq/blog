# 链接查询

7种链接查询

# 左连接（A独有+AB共有）

A left join B on

select t1.id,t1.deptName,count(t2.name) as emps from tbl_dept t1 left join tbl_emp t2 
on t2.deptId=t1.id group by deptName order by id;

# 右连接（B独有+AB共有）

A right join B on

select t2.id,t2.name,t1.deptName from tbl_dept t1 right join tbl_emp t2 
on t2.deptId=t1.id;

# 内连接 （AB共有）

A inner join B on

select deptName,t2.name empName from tbl_dept t1 inner join tbl_emp t2 
on t1.id=t2.deptId;

# A独有

A inner join B on where b.id is null

select a.deptName,b.name empName from tbl_dept a left join tbl_emp b 
on a.id=b.deptId where b.id is null;

# B独有

A left join B on where a.id is null

select a.name empName,b.deptName from tbl_emp a left join tbl_dept b
on a.deptId=b.id where a.id is null;

# A独有+B独有

A left join B on where b.id is null union A left join B on where a.id is null

select a.deptName,b.name empName from tbl_dept a left join tbl_emp b 
on a.id=b.deptId where b.id is null union select b.deptName,a.name emptName from tbl_emp a left join tbl_dept b on a.deptId=b.id where a.id is null;

# A独有+AB公共+B独有

A left join B on a.id=b.deptId union A left join B on a.id=b.deptId

select a.deptName,b.name empName  from tbl_dept a left join tbl_emp b
on a.id=b.deptId union select a.deptName,b.name empName from tbl_dept a right join tbl_emp b on a.id=b.deptId;

TIP

所有的join返回的结果都是倍数的

# 存储过程

# IN 输入参数

表示该参数的值必须在调用存储过程时指定，在存储过程中修改该参数的值不能被返回，为默认值

# OUT 输出参数

该值可在存储过程内部被改变，并可返回

# INOUT 输入输出参数

调用时指定，并且可被改变和返回

# 事务transaction

事务可以保证多个操作原子性，对于数据库来说，事务可以保证批量操作要么全成功，要么全失败。

# 事务具有四个特征ACID

原子性（Atomicity），事务是最小单元，不可再分；

一致性（Consistency），事务要求所有的DML语句操作的时候，必须保证同时成功或同时失败；

隔离性（Isolation），一个事务不会影响其他事务的执行；

持久性（Durability），在事务完成之后，该事务对数据库所作的更改将持久地保存在数据库中，并不会被回滚；

# MySQL事务的提交和回滚

MySQL中默认情况下，事务是自动提交的，当执行一条DML语句时，就开启并且自动提交了事务。可以通过下面的语句查看事务是否是自动提交的：

show variables like '%commit%';

如果想要关闭MySQL事务的自动提交，可以使用下面语句手动开启事务：

start transaction;

回滚操作 rollback; 这时候相关的数据修改都会还原回来

提交事务 commit; 当开开启事务需要自己手动commit才能修改生效

IF (tmp_mb>0) THEN
	SET gpt_state = 1;
ELSE
	#开启事务 
	START TRANSACTION; 
	insert into 
	auth_user(uid, user_name,
	mobile, nick, pass_word, user_status,
	province, city, area, create_time, 
	create_user_name, create_user_uid,partner_id)
	value
	(REPLACE(UUID(), '-', ''),pt_un,pt_mb,pt_nick,pt_pw,
	pt_us,pt_province,pt_city,pt_area,NOW(),
	pt_cun,pt_cuu,pt_partner);
	#提交事务或者回滚
	IF t_error = 1 THEN 
			ROLLBACK;
			set gpt_state = -5;
	ELSE 
			COMMIT;
			SELECT COUNT(*) into gpt_state from auth_user where user_name = pt_un and user_status <> -1;
	END IF; 
END IF;

# 关联表里有重复字段

-- 内连接降序
select student_id ,student_name,course_name,score 
from student_info a JOIN score_info b JOIN course_info c
on a.student_id=b.student_id and c.course_name='物流管理'
and b.course_id=c.course_id ORDER BY a.student_id DESC;

报错:Column 'student_id' in field list is ambiguous 意思是关联表里有重复字段

解决:前面加上前缀就没问题了

select a.student_id

# on

join 和 inner join 效果是一样，但right join 和 left join都必须写on 条件不然会报错

SELECT * FROM course_info  RIGHT JOIN  score_info on 1=1;

# where和HAVING的区别

  1. GROUP BY ORDER BY 不能放where前面

  2. where后面不能用聚合函数:如SUM，count等，如要只能用HAVING条件

  3. having是分组（group by）后的筛选条件，如果你没写分组直接having 的也会默认分组

# 分组的概念

分组的概念是把相同合并成一条记录（所以每一组只有一条记录）其他字段值取第一条的

SELECT *,count(address) FROM student_info  GROUP BY address HAVING COUNT(address)>0;
