# namespace

控制mapper对应接口作用域

<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE mapper
    PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN"
    "http://mybatis.org/dtd/mybatis-3-mapper.dtd">
<mapper namespace="com.lzq.o2o.dao.ShopDao">
    <!--sql标签写在里面-->
</mapper>

# useGeneratedKeys

insert里可以返回主键,可以省去再次去查询

<insert id="insertShop" useGeneratedKeys="true" keyColumn="shop_id" keyProperty="shopId">
    <!--sql-->
</insert>

int insertShop = shopDao.insertShop(shop);
System.out.println(insertShop);//1 表示成功
System.out.println(shop.getShopId()); //66 ，新插入的shop_id值直接映射到shop对象，省去再去查询。

# parameterType

  • select、insert、update、delete元素中有一个parameterType属性，用于对应的mapper接口方法接受的参数类型
  • 一般会根据参数自动识别所以不写也可以

# 参数类型

  • 基本数据类型：int,string,long等
  • 复杂数据类型：类和Map等

# 获取参数中的值

  • 基本数据类型：#{参数}
  • 复杂数据类型：#{属性名} ， map中则是#{key}
<update id="updateShop" parameterType="com.lzq.o2o.entity.Shop">
    UPDATE tb_shop
    <set>
        <if test="shopName!=null">shop_name=#{shopName},</if>
        <if test="area!=null">area_id=#{area.areaId}</if>
    </set>
    where shop_id=#{shopId}
</update>

# @Param 注解

类似于将参数重命名, 当多个参数的时候必须要去指定名字

List<String> selectIdByOpenType(@Param(value="open")String openType);

<select id="selectIdByOpenType"  parameterType="java.lang.String"> 
     select a ,b ,c  where open = 123
 </select>

# where标签

当if条件不成力的时候回生成字符串where 1=1

<select id="selectWhere" resultType="student">
    SELECT id,name,age,score
    FROM t_student
    <where>
        <if test="name != null and name != ''">
            name LIKE '%' #{name} '%'
        </if>
    </where>
</select>

# choose标签

在choose标签中可以有多个when，但是只能有一个otherwise，这个有点类似java中的switch语句。

<select id="selectChoose" resultType="student">
    SELECT id,name,age,score
    FROM t_student
    <where>
        <choose>
              <when test="name != null and name != ''">
                  name LIKE '%' #{name} '%'
              </when>
              <when test="age>=0">
                  age > #{age}
              </when>
              <otherwise>
                  1 != 1
              </otherwise>
        </choose>
    </where>
</select>

</select>

# sql/include标签

sql标签可以用来定义一个可被复用的sql片段，在使用的时候写上include标签就可以将sql标签中的内容引入

<!--定义sql片段-->
<sql id="select">
    SELECT id,name,age,score
    FROM t_student
</sql>

<select id="selectSQL" resultType="student">
    <!--使用sql片段-->
    <include refid="select"/>

    <if test="list != null and list.size>0">
        WHERE id IN
        <foreach collection="list" open="(" close=")" item="stu" separator=",">
            #{stu.id}
        </foreach>
    </if>
</select>

# trim 标签

可以更灵活的组装sql字符串

  • prefix:加上前缀
  • suffix:加上后缀
  • suffixOverrides:覆盖后缀
  • prefixOverrides:覆盖前缀
<trim prefix="" suffix="" suffixOverrides="" prefixOverrides=""></trim>

示例：模仿where标签

select * from user 
<trim prefix="WHERE" prefixoverride="AND |OR">
　　<if test="name != null and name.length()>0"> AND name=#{name}</if>
　　<if test="gender != null and gender.length()>0"> AND gender=#{gender}</if>
</trim>

<select id="queryByShopId" resultMap="shopMap" parameterType="Long">
    SElECT
    s.shop_id,
    a.area_id,
    sc.shop_category_id,
    FROM
    tb_shop s,
    tb_area a,
    tb_shop_category sc
    WHERE
    s.area_id=a.area_id

    s.shop_category_id=sc.shop_category_id

    s.shop_id=#{shopId}

# 自关联查询

自关联查询就是自己充当多方，可以看做是一个树形结构，表中有一个外键示当前数据的父节点。

<resultMap id="leaderMap" type="Employee">
    <id column="id" property="id"/>
    <result column="name" property="name"/>
    <association property="leader" javaType="Employee" column="mgr" select="selectLeaderByPid" />
</resultMap>

<select id="selectLeaderByPid" resultMap="leaderMap">
    SELECT id,name,job,mgr
    FROM t_employee
    WHERE id=#{id}
</select>

# 比较符

实体符号

对一些数据进行比较大于>、小于< 就有可能导致xml文件解析出现问题，此时可以使用实体符号代替。

原符号      实体符号
  <         &lt;
  <=        &lt;=
  >         &gt;
  >=        &gt;=
  &         &amp;
  "         &quot;
  '         &apos;

CDATA

还可以将这些数据放到里面，这里面的内容xml是不会解析的。

<select id="selectIf" resultType="student">
    SELECT id,name,age,score
    FROM t_student
    WHERE 1=1
    <if test="name != null and name != ''">
      AND name LIKE '%' #{name} '%'
    </if>
    <if test="age>=0">
        AND age <![CDATA[  > ]]> #{age}
    </if>
</select>

# Mapper中的方法不能重载

mybatis中mapper.xml是不会准确映射到Java中的重载方法的。最好不要在mapper接口中使用方法重载。

//错误示例：
public interface ShopDao {
	int updateShop(Shop shop);
	int updateShop(Shop shop,int a);
}

# 查询缓存

  • 查询缓存主要是为了提高查询访问速度，即当用户执行一次查询后，会将该数据结果放到缓存中， 当下次再执行此查询时就不会访问数据库了而是直接从缓存中获取该数据。

  • 查询缓存的作用域是根据映射文件 mapper 的 namespace 划分的，相同namespace 的 mapper 查询数据存放在同一个缓存区域

# 一级缓存

也叫作本地缓存， 默认一级查询缓存是开启状态，且不能关闭

# 二级缓存
