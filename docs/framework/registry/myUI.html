# 创建ui项目

  1. 用vue-cli创建个项目
  2. 创建个包packages来存放需要发布的组件
  3. npm init初始化个package.json
{
  "name": "myUI",
  "version": "0.0.1",
  "description": "UI base on Vue",
  "main": "index.js",
  "scripts": {
    "test": "echo \"Error: no test specified\" && exit 1"
  },
  "keywords": [
    "UI"
  ],
  "author": "LZQ",
  "license": "ISC",
  "homepage": ""
}
  1. 创建个index.js作为默认加载文件,来用汇总导出组件 index.js
import Vue from 'vue'
import upLoad from './upLoad'
import phoneVcode from './phoneVcode'
import inputSearch from './inputSearch'
import formatMoney from './formatMoney'

const components = [
  upLoad,
  phoneVcode,
  inputSearch,
  formatMoney
];

// vue.use()会加载这个函数，一般我们把所有组件都挂上去
const install = function (Vue) {
  components.map(component => Vue.component(component.name, component))
};

export default {
  //全部组件对象 
  install,
  // 按需组件对象
  upLoad,
  phoneVcode,
  inputSearch,
  formatMoney
}

我们也可以创建多个js文件来导出，如common.js

import Vue from 'vue'

let _myCommonJS = {
  //手机号验证
  checkPhone(phone) {
    return (/^[1][3,4,5,7,8,9][0-9]{9}$/.test(phone)) ? true : false;
  }
};

// vue.use()会记载这个函数
const  install = function (Vue, options) {
  Object.defineProperties(Vue.prototype, {
    _myCommonJS: {
      get() {
        return _myCommonJS;
      }
    }
  });
};

export default {
  install,
}

# npm发布你的UI框架

  1. package.json里的版本号不能是仓库已有的，所以我们要递增修改 version": "0.8.4",
  2. npm发布

# 使用

安装

npm install myUI --D

引用

import Vue from 'vue'
import myUI from 'myUI'
Vue.use(myUI)

import myCommonJS from 'myUI/common.js'
Vue.use(myUImyCommonJS)

常用命令

更新到最后一版

npm i myUI@latest -D

更新指定版本

npm i myUI@0.0.6 -D
